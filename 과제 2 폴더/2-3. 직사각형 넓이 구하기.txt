#include <string>
#include <vector>

using namespace std;

// 제한사항
bool check_dots(vector<vector<int>> dots)
{
    bool result = false;
    
    //dots의 길이 = 4
    if(dots.size() == 4)
    {
        result = true;
    }
    
    for(vector<int> line : dots)
    {
        //dots의 원소의 길이 = 2
        if(line.size() == 2)
        {
            result = true;
        }
        else
        {
            result = false;
            break;
        }
        //-256 < dots[i]의 원소 < 256
        for(int dot : line)
        {
            if((-256 < dot) && (dot < 256))
            {
                result = true;
            }
            else
            {
                result = false;
                break;
            }
        }
    }
    return result;
}

int solution(vector<vector<int>> dots) {
    int answer = 0;
    int x_max = 0, x_min = 0;
    int y_max = 0, y_min = 0;

    if(check_dots(dots))
    {
        int x = 0;
        int y = 0;
        for(vector<int> line : dots)
        {
            if(x == 0)
            {
                x = line[0];
                y = line[1];
            }
            else
            {
                if(x == line[0])
                {
                    if(y < line[1])
                    {
                        y_min = y;
                        y_max = line[1];
                    }
                    else
                    {
                        y_max = y;
                        y_min = line[1];
                    }
                }
                else
                {
                    if(x < line[0])
                    {
                        x_min = x;
                        x_max = line[0];
                    }
                    else
                    {
                        x_min = line[0];
                        x_max = x;
                    }
                }
            }
        }
    }
    
    answer = (x_max - x_min) * (y_max - y_min);
    return answer;
}
